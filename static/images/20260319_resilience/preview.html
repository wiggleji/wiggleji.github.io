01 Rate Limiter 실행 전 — 순간 판단

초당/주기당 허용 요청 수를 제한합니다. 통과하면 Rate Limiter의 역할은 즉시 끝납니다 — 실행 중에는 아무것도 점유하지 않습니다.

limitForPeriod: 5limitRefreshPeriod: 1stimeoutDuration: 300ms
yamlapplication.yml
resilience4j:
  ratelimiter:
    instances:
      kakaoApi:
        limitForPeriod: 100        # 주기당 허용 횟수
        limitRefreshPeriod: 1s   # 1초마다 quota 리셋
        timeoutDuration: 500ms    # 슬롯 대기 시간 (초과 시 예외)
        registerHealthIndicator: true
        eventConsumerBufferSize: 100
kotlinannotation — NotificationService.kt
@Service
class NotificationService {

    // @RateLimiter: 설정된 name의 인스턴스를 적용, fallbackMethod는 선택 사항
    // 같은 클래스 내에 위치해야 하며 시그니처는 원본과 동일 + 예외 파라미터 1개
    @RateLimiter(name = "kakaoApi", fallbackMethod = "rateFallback")
    fun sendMessage(userId: String, msg: String): String {
        return kakaoClient.send(userId, msg)
    }

    // fallback 시그니처: 원본 파라미터 + 예외 타입 (가장 구체적인 것 우선 매칭)
    private fun rateFallback(userId: String, msg: String, ex: RequestNotPermittedException): String {
        log.warn("Rate limit exceeded for {}: {}", userId, ex.message)
        kafkaProducer.send("notify.retry", NotifyEvent(userId, msg))
        return "queued"
    }

    // 전역 fallback: Exception 타입으로 모든 예외를 한번에 처리
    private fun rateFallback(userId: String, msg: String, ex: Exception): String {
        return "error"
    }
}
kotlinclass-based — RateLimiterRegistry 직접 사용
@Service
class NotificationService(
    private val rateLimiterRegistry: RateLimiterRegistry,  // Spring이 자동 주입
    private val kakaoClient: KakaoClient
) {
    private val rateLimiter = rateLimiterRegistry.rateLimiter("kakaoApi")

    fun sendMessage(userId: String, msg: String): String {
        // RateLimiter.decorateSupplier: Supplier를 래핑해 quota 초과 시 예외 발생
        val decorated = RateLimiter.decorateSupplier(rateLimiter) {
            kakaoClient.send(userId, msg)
        }
        return Try.ofSupplier(decorated)
            .recover(RequestNotPermittedException::class.java) { ex ->
                log.warn("Rate limited: {}", ex.message)
                kafkaProducer.send("notify.retry", NotifyEvent(userId, msg))
                "queued"
            }.get()
    }

    fun sendMessageDirect(userId: String, msg: String): String {
        // acquirePermission: blocking 방식으로 허가 대기 (timeoutDuration 적용)
        rateLimiter.acquirePermission()
        return kakaoClient.send(userId, msg)
    }

    @PostConstruct
    fun setupEventListeners() {
        // 이벤트 리스닝으로 quota 소진 모니터링
        rateLimiter.eventPublisher
            .onFailure { log.warn("Rate limit failure: {}", it) }
            .onSuccess { log.debug("Rate limit success") }

        // 런타임 설정 변경 (동적 조정 가능)
        rateLimiter.changeLimitForPeriod(200)  // 핫 변경
    }
}
Multi Instance 주의 — Rate Limiter는 인스턴스 로컬입니다. 3개 인스턴스 × limitForPeriod 100 = 실제로 300 요청이 나갑니다. 외부 API quota 보호가 목적이라면 Redis + Lua 스크립트로 중앙화해야 합니다.
02 Bulkhead 전 획득 → 중 점유 → 후 반환

동시 실행 수를 제한합니다. 슬롯을 실행 전에 잡고 실행이 끝나야 반환하므로, 외부 서비스가 느릴수록 슬롯이 빠르게 소진됩니다.

yamlapplication.yml
resilience4j:
  # Semaphore — 동기 호출 (WebMVC)
  bulkhead:
    instances:
      inventoryService:
        maxConcurrentCalls: 20   # 동시 최대 20개
        maxWaitDuration: 100ms  # 슬롯 대기 시간

  # ThreadPool — 비동기 호출 (TimeLimiter와 함께)
  thread-pool-bulkhead:
    instances:
      reportService:
        coreThreadPoolSize: 4
        maxThreadPoolSize: 8
        queueCapacity: 20
        writableStackTraceEnabled: true
kotlinannotation — BulkheadService.kt
@Service
class InventoryService {

    // Semaphore (기본값): 동기 메서드에 적용. maxConcurrentCalls 초과 시 즉시 예외
    @Bulkhead(name = "inventoryService", fallbackMethod = "stockFallback")
    fun checkStock(itemId: String): StockResponse =
        inventoryClient.check(itemId)

    private fun stockFallback(itemId: String, ex: BulkheadFullException): StockResponse =
        StockResponse(available = false, message = "서버 과부하, 잠시 후 재시도")
}

@Service
class ReportService {

    // ThreadPool: 비동기 전용. 전용 스레드풀에서 실행 → 메인 풀 보호
    // type = THREADPOOL 이면 반드시 CompletableFuture 반환해야 함
    @Bulkhead(name = "reportService", type = Bulkhead.Type.THREADPOOL, fallbackMethod = "reportFallback")
    @TimeLimiter(name = "reportService")
    fun generate(req: ReportRequest): CompletableFuture<Report> =
        CompletableFuture.supplyAsync { reportRepository.heavyQuery(req) }

    private fun reportFallback(req: ReportRequest, ex: Exception): CompletableFuture<Report> =
        CompletableFuture.completedFuture(Report(status = "UNAVAILABLE"))
}
kotlinclass-based — BulkheadRegistry 직접 사용
@Service
class InventoryService(
    private val bulkheadRegistry: BulkheadRegistry,
    private val threadPoolBulkheadRegistry: ThreadPoolBulkheadRegistry
) {
    private val semaphoreBulkhead = bulkheadRegistry.bulkhead("inventoryService")
    private val threadPoolBulkhead = threadPoolBulkheadRegistry.bulkhead("reportService")

    // Semaphore: decorateSupplier로 래핑
    fun checkStock(itemId: String): StockResponse {
        val decorated = Bulkhead.decorateSupplier(semaphoreBulkhead) {
            inventoryClient.check(itemId)
        }
        return Try.ofSupplier(decorated)
            .recover(BulkheadFullException::class.java) {
                StockResponse(available = false)
            }.get()
    }

    // ThreadPool: decorateSupplier → CompletableFuture 반환
    fun generateAsync(req: ReportRequest): CompletableFuture<Report> {
        val decorated = ThreadPoolBulkhead.decorateSupplier(threadPoolBulkhead) {
            reportRepository.heavyQuery(req)
        }
        return decorated.get()
            .exceptionally { Report(status = "UNAVAILABLE") }
    }

    @PostConstruct
    fun monitorBulkhead() {
        // 현재 사용 중인 슬롯 수 조회
        val metrics = semaphoreBulkhead.metrics
        log.info("Available permits: {}", metrics.availableConcurrentCalls)
        log.info("Max permits: {}", metrics.maxAllowedConcurrentCalls)

        semaphoreBulkhead.eventPublisher
            .onCallPermitted  { log.debug("Call permitted, remaining: {}", it) }
            .onCallRejected   { log.warn("Call rejected — bulkhead full") }
            .onCallFinished   { log.debug("Call finished, slot released") }
    }
}
maxConcurrentCalls: 3maxWaitDuration: 200ms
TimeLimiter와 함께 써야 하는 이유 — 외부 서비스가 응답을 10초씩 지연하면 슬롯 20개가 순식간에 소진됩니다. ThreadPool Bulkhead + TimeLimiter를 조합해 슬롯 점유 시간 자체를 제한하세요.
03 Circuit Breaker 전 차단 + 후 기록 (두 시점)

가장 핵심적인 패턴. CLOSED에서는 결과를 기록하고, 실패율이 임계값을 넘으면 OPEN으로 전환해 호출을 차단합니다.

slidingWindowSize: 10failureRateThreshold: 50%waitDurationInOpenState: 10spermittedCallsInHalfOpen: 3
yamlapplication.yml
resilience4j:
  circuitbreaker:
    configs:
      default:
        slidingWindowType: COUNT_BASED  # 또는 TIME_BASED
        slidingWindowSize: 10
        failureRateThreshold: 50          # 50% 실패 시 OPEN
        slowCallDurationThreshold: 3000   # 3초 이상 = slow
        slowCallRateThreshold: 80
        waitDurationInOpenState: 10000   # OPEN 유지 10초
        permittedNumberOfCallsInHalfOpenState: 3
        minimumNumberOfCalls: 5
        registerHealthIndicator: true
        recordExceptions:
          - java.io.IOException
          - java.util.concurrent.TimeoutException
        ignoreExceptions:
          - com.example.BusinessException  # 비즈니스 예외는 카운트 제외
      strict:                                # 결제 등 민감한 서비스용
        baseConfig: default
        failureRateThreshold: 30
        waitDurationInOpenState: 30000
    instances:
      paymentService:
        baseConfig: strict
      inventoryService:
        baseConfig: default
kotlinannotation — PaymentService.kt
@Service
class PaymentService(private val client: PgClient) {

    // fallbackMethod: OPEN 상태이거나 recordExceptions에 해당하는 예외 발생 시 실행
    // 예외 타입별로 여러 fallback 정의 가능 — 가장 구체적인 타입이 우선 매칭됨
    @CircuitBreaker(name = "paymentService", fallbackMethod = "paymentFallback")
    fun charge(req: PaymentRequest): PaymentResponse =
        client.charge(req)

    // CallNotPermittedException: OPEN 상태에서 차단된 경우
    private fun paymentFallback(req: PaymentRequest, ex: CallNotPermittedException): PaymentResponse {
        log.warn("CB OPEN — payment blocked")
        return PaymentResponse(status = PENDING, message = "결제 시스템 점검 중")
    }

    // Exception: 그 외 recordExceptions에 해당하는 모든 예외
    private fun paymentFallback(req: PaymentRequest, ex: Exception): PaymentResponse {
        log.warn("CB fallback: {}", ex.message)
        return PaymentResponse(status = PENDING, message = "결제 대기 중")
    }
}
kotlinclass-based — CircuitBreakerRegistry 직접 사용
@Service
class PaymentService(
    private val circuitBreakerRegistry: CircuitBreakerRegistry,
    private val pgClient: PgClient
) {
    private val cb = circuitBreakerRegistry.circuitBreaker("paymentService")

    fun charge(req: PaymentRequest): PaymentResponse {
        // decorateSupplier: Supplier를 CB로 래핑
        // executeSupplier: 즉시 실행 (shorthand)
        val decorated = CircuitBreaker.decorateSupplier(cb) { pgClient.charge(req) }
        return Try.ofSupplier(decorated)
            .recover(CallNotPermittedException::class.java) {
                PaymentResponse(status = PENDING)
            }
            .recover(Exception::class.java) { ex ->
                log.warn("CB error: {}", ex.message)
                PaymentResponse(status = PENDING)
            }.get()
    }

    @PostConstruct
    fun setupMonitoring() {
        // 상태 조회
        log.info("CB state: {}", cb.state)
        log.info("Failure rate: {}%", cb.metrics.failureRate)
        log.info("Buffered calls: {}", cb.metrics.numberOfBufferedCalls)

        // 상태 전환 이벤트 구독
        cb.eventPublisher
            .onStateTransition { e ->
                log.warn("CB [{}] {} → {}", e.circuitBreakerName,
                    e.stateTransition.fromState, e.stateTransition.toState)
            }
            .onFailureRateExceeded { e ->
                log.warn("Failure rate exceeded: {}%", e.failureRate)
            }
            .onCallNotPermitted { log.warn("Call blocked — CB is OPEN") }

        // 런타임 강제 상태 변경 (테스트/운영 비상용)
        // cb.transitionToOpenState()
        // cb.transitionToClosedState()
    }
}
04 Time Limiter 실행 중 — Future 감시

CompletableFuture(비동기) 전용입니다. 동기 메서드에는 단독으로 동작하지 않습니다. ThreadPool Bulkhead와 함께 사용합니다.

timeoutDuration: 3scancelRunningFuture: true
yamlapplication.yml
resilience4j:
  timelimiter:
    instances:
      reportService:
        timeoutDuration: 3s
        cancelRunningFuture: true   # 타임아웃 시 Future 취소

  thread-pool-bulkhead:            # 함께 쓰는 것이 일반적
    instances:
      reportService:
        coreThreadPoolSize: 4
        maxThreadPoolSize: 8
        queueCapacity: 20
kotlinannotation — ReportService.kt
@Service
class ReportService {

    // @TimeLimiter는 반드시 CompletableFuture 반환 메서드에만 동작
    // ThreadPool Bulkhead와 함께 써야 전용 스레드풀에서 비동기 실행됨
    // fallbackMethod 시그니처는 원본과 동일 + TimeoutException 파라미터
    @Bulkhead(name = "reportService", type = Bulkhead.Type.THREADPOOL)
    @TimeLimiter(name = "reportService", fallbackMethod = "reportFallback")
    fun generate(req: ReportRequest): CompletableFuture<Report> =
        CompletableFuture.supplyAsync {
            reportRepository.heavyQuery(req)  // 3초 초과 시 cancel
        }

    // cancelRunningFuture=true 이면 Future.cancel(true) 호출 → 스레드 인터럽트 시도
    private fun reportFallback(req: ReportRequest, ex: TimeoutException)
        : CompletableFuture<Report> =
        CompletableFuture.completedFuture(
            Report(status = "TIMEOUT", message = "보고서 생성 시간 초과")
        )
}
kotlinclass-based — TimeLimiterRegistry 직접 사용
@Service
class ReportService(
    private val timeLimiterRegistry: TimeLimiterRegistry,
    private val executor: ScheduledExecutorService
) {
    private val timeLimiter = timeLimiterRegistry.timeLimiter("reportService")

    fun generate(req: ReportRequest): Report {
        // executeCompletionStage: CompletableFuture를 timeoutDuration으로 래핑
        // 내부적으로 ScheduledExecutorService에 타임아웃 작업을 예약함
        val future: Callable<CompletableFuture<Report>> = Callable {
            CompletableFuture.supplyAsync { reportRepository.heavyQuery(req) }
        }
        return Try.of(TimeLimiter.decorateFutureSupplier(timeLimiter, executor, future))
            .recover(TimeoutException::class.java) {
                Report(status = "TIMEOUT")
            }.get()
    }

    @PostConstruct
    fun monitorTimeLimiter() {
        timeLimiter.eventPublisher
            .onTimeout  { log.warn("TimeLimiter timeout after {}s",
                timeLimiter.timeLimiterConfig.timeoutDuration.seconds) }
            .onSuccess  { log.debug("TimeLimiter success") }
            .onError    { log.warn("TimeLimiter error: {}", it) }
    }
}
05 Retry 실행 후 — 예외 확인 후 재호출

Method()가 예외를 던진 후에만 개입합니다. 성공하면 즉시 루프를 탈출합니다.

maxAttempts: 3waitDuration: 1sexpBackoffMultiplier: 2
yamlapplication.yml
resilience4j:
  retry:
    instances:
      externalApi:
        maxAttempts: 3                   # 최초 1회 + 재시도 2회
        waitDuration: 1000               # 기본 대기 1초
        enableExponentialBackoff: true
        exponentialBackoffMultiplier: 2   # 1s → 2s → 4s
        exponentialMaxWaitDuration: 10000
        retryExceptions:
          - java.io.IOException
          - feign.RetryableException
        ignoreExceptions:
          - com.example.BadRequestException  # 4xx는 재시도 무의미
          - com.example.BusinessException
kotlinannotation — ExternalApiService.kt
@Service
class ExternalApiService {

    // @Retry는 retryExceptions에 해당하는 예외 발생 시 재시도
    // ignoreExceptions에 해당하면 즉시 예외 전파 (재시도 없음)
    // CB와 조합 시: Retry가 안쪽 → 재시도 모두 실패한 결과 1개를 CB가 카운트
    @CircuitBreaker(name = "externalApi", fallbackMethod = "apiFallback")
    @Retry(name = "externalApi")
    fun call(req: ApiRequest): ApiResponse =
        httpClient.post(req)

    // fallback: 예외 타입별로 여러 개 정의 가능 (가장 구체적인 타입 우선 매칭)
    private fun apiFallback(req: ApiRequest, ex: CallNotPermittedException): ApiResponse =
        ApiResponse(cached = true, data = cache.get(req.key), reason = "CB OPEN")

    private fun apiFallback(req: ApiRequest, ex: Exception): ApiResponse =
        ApiResponse(cached = true, data = cache.get(req.key))

    // retryOnResult: 응답 내용 기반 재시도 (예외 없이 잘못된 응답일 때)
    @Retry(name = "externalApi")
    fun callWithResultCheck(req: ApiRequest): ApiResponse {
        val resp = httpClient.post(req)
        // retryOnResultPredicate는 yaml로 설정하거나 Customizer로 주입
        if (resp.status == 503) throw RetryableException("503 received")
        return resp
    }
}
kotlinclass-based — RetryRegistry 직접 사용
@Service
class ExternalApiService(
    private val retryRegistry: RetryRegistry,
    private val circuitBreakerRegistry: CircuitBreakerRegistry
) {
    private val retry = retryRegistry.retry("externalApi")
    private val cb    = circuitBreakerRegistry.circuitBreaker("externalApi")

    fun call(req: ApiRequest): ApiResponse {
        // 함수형 체이닝: CB로 먼저 감싸고 Retry로 한 번 더 감쌈
        // 실행 순서: Retry → (재시도마다) CB 체크 → Method → 결과를 CB에 기록
        val decorated = Decorators.ofSupplier { httpClient.post(req) }
            .withCircuitBreaker(cb)
            .withRetry(retry)
            .withFallback(
                listOf(CallNotPermittedException::class.java, MaxRetriesExceededException::class.java)
            ) { ex -> ApiResponse(cached = true, data = cache.get(req.key)) }
            .decorate()
        return decorated.get()
    }

    fun callWithResultRetry(req: ApiRequest): ApiResponse {
        // retryOnResult: 예외 없이 결과 자체가 재시도 조건인 경우
        val retryOnResult = Retry.decorateSupplier(retry) { httpClient.post(req) }
        return Try.ofSupplier(retryOnResult)
            .recover(MaxRetriesExceededException::class.java) {
                ApiResponse(cached = true)
            }.get()
    }

    @PostConstruct
    fun monitorRetry() {
        retry.eventPublisher
            .onRetry  { e -> log.warn("Retry attempt #{} after: {}",
                e.numberOfRetryAttempts, e.lastThrowable?.message) }
            .onSuccess { log.info("Retry succeeded on attempt #{}",
                it.numberOfRetryAttempts) }
            .onError   { log.warn("All retries exhausted: {}", it.lastThrowable?.message) }
            .onIgnoredError { log.debug("Ignored exception, no retry: {}", it) }

        // 메트릭 조회
        log.info("Retry success calls: {}",    retry.metrics.numberOfSuccessfulCallsWithoutRetryAttempt)
        log.info("Retry retried calls: {}",   retry.metrics.numberOfSuccessfulCallsWithRetryAttempt)
        log.info("Retry failed calls: {}",    retry.metrics.numberOfFailedCallsWithoutRetryAttempt)
        log.info("Retry exhausted calls: {}", retry.metrics.numberOfFailedCallsWithRetryAttempt)
    }
}
결제 API 재시도 주의 — idempotency-key 없이 재시도하면 이중 결제가 발생합니다. 결제는 maxAttempts: 2 이하로 제한하고, PG사의 멱등성 지원 여부를 반드시 확인하세요.

07 Multi Instance (K8s)

Scale-out 환경에서 모듈별 동작 차이를 정리합니다.

Circuit Breaker
인스턴스별 독립
외부 장애 시 수초 내 자연 수렴. 대부분 허용 가능.
Rate Limiter
Redis 중앙화 필수
인스턴스 3개면 실제 3배 요청. 외부 quota 보호 불가.
Bulkhead
인스턴스별 독립
로드밸런서가 분산하므로 자연스럽게 나뉨.
Retry
인스턴스별 독립
각 인스턴스가 독립적으로 재시도. 문제 없음.
kotlinRedisRateLimiter.kt — Multi Instance용
@Component
class RedisRateLimiter(private val redis: StringRedisTemplate) {

    private val script = DefaultRedisScript<Long>().apply {
        setScriptText("""
            local current = redis.call('INCR', KEYS[1])
            if current == 1 then
                redis.call('EXPIRE', KEYS[1], ARGV[2])
            end
            if current > tonumber(ARGV[1]) then return 0 end
            return 1
        """)
        resultType = Long::class.java
    }

    fun tryAcquire(key: String, limit: Int, windowSec: Int): Boolean =
        redis.execute(script, listOf(key), limit.toString(), windowSec.toString()) == 1L
}
